  2. How many parameters are there in the above model?
  3. Write the code to add a legend to the following graph.

def model1(f_0,f_1,a,b,T):
    f_values = [f_0,f_1]
    for t in range(1,T):
        f_new = a*f_values[-1] - b*f_values[-2]
        f_values.append(f_new)
    return f_values

def graph_model1(f_0,f_1,a,b,T):
    inputs = [t for t in range(T+1)]
    outputs = model1(f_0,f_1,a,b,T)
    plt.plot(inputs,outputs,'.')
    plt.title('model1 for a={}, b={}, f_0={}, f_1={} over [0,{}]'.format(a,b,f_0,f_1,T))
    plt.xlabel('t')
    plt.ylabel('f(t)')
    plt.show()

def graph_model2(f_0,f_1,list_of_a,b,T):
    inputs = [t for t in range(T+1)]
    list_of_labels = []
    for a in list_of_a:
        outputs = model1(f_0,f_1,a,b,T)
        list_of_labels.append('a={}'.format(a))
        plt.plot(inputs,outputs)
    plt.title('model1 for b={}, f_0={}, f_1={} over [0,{}]'.format(b,f_0,f_1,T))
    plt.xlabel('t')
    plt.ylabel('f(t)')
    plt.legend(list_of_labels)
    plt.show()

def model_bifurcation(f_0,f_1,b,T):
    a_values = [a/100 for a in range(101)]
    f_T_values = [model1(f_0,f_1,a,b,T)[-1] for a in a_values]
    plt.plot(a_values,f_T_values)
    plt.title('a-value bifurcation diagram with f_0={}, f_1={}, b={} over [0,{}]'.format(f_0,f_1,b,T))
    plt.xlabel('value of the a parameter')
    plt.ylabel('f(T)')
    plt.show()
