def plot_sin():
    from math import sin,pi
    inputs = [2*pi*i/100 for i in range(101)]
    outputs = [sin(x) for x in inputs]
    plt.plot(inputs,outputs)
    plt.title('$y=\sin(x)$')
    plt.xlabel('$x$')
    plt.ylabel('$y$')
    plt.xticks([0,pi/2,pi,3*pi/2,2*pi],['0','$\pi/2$','$\pi$','$3\pi/2$','$2\pi$'])
    plt.yticks([-1,0,1],['-1','0','1'])
    plt.show()
