def add_it_up(N):              # Define the function with input N, the number of numbers to add up
    sum1 = 0                   # Initialize the variable we will use to store the sum
    for num in range(1,N+1):   # For each number in the range 1,2,3,...,N
        sum1 += num            # Add that number to the total stored in the variable sum1
    return sum1                # Return the output

def Fibonacci(length):                                                             # Input is the number of desired terms
    list_of_fibonacci_numbers = [1,1]                                              # Initialize the list with first two
    for i in range(2,length):                                                      # You can also use range(0,length-2)
        next_number = list_of_fibonacci_numbers[-1] + list_of_fibonacci_numbers[-2]# Calculate the next fibonacci number
        list_of_fibonacci_numbers.append(next_number)                              # Add the new number to the list
    return list_of_fibonacci_numbers                                               # Return the list of numbers
