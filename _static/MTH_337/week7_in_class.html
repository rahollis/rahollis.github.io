def entry_sum(arr):
    summation = 0
    r,c = arr.shape
    for i,j in [(i,j) for i in range(r) for j in range(c)]:
        summation += arr[i,j]
    return summation

def entry_sum1(arr):
    summation = 0
    for row in arr:
        for entry in row:
            summation += entry
    return summation

def middle_ones(n,k):
    arr = np.zeros((n,n),dtype=int)
    arr[(n-k)//2:(n+k)//2,(n-k)//2:(n+k)//2] = 1
    return arr
