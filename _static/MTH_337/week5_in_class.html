Population models¶

$$ P(t+1) = k\cdot P(t) $$

def pop_model(init_pop,growth_factor):
    pop_vals = [init_pop]
    for t in range(10):
        new_val = growth_factor*pop_vals[t]
        pop_vals.append(new_val)
    return pop_vals

def plot_model(init_pop,growth_factor):
    x = [t for t in range(11)]
    y = pop_model(init_pop,growth_factor)
    plt.figure(figsize=[10,8])
    plt.plot(x,y)
    plt.title('Plot of $P(t)$ with initial pop {} and growth factor {}'.format(init_pop,growth_factor))
    plt.xlabel('Time in years')
    plt.ylabel('Population in thousands')

def plot_models(init_pop,growth_factors):
    plt.figure(figsize=[10,8])
    for gf in growth_factors:
        x = [t for t in range(11)]
        y = pop_model(init_pop,gf)
        plt.plot(x,y,label='growth factor {}'.format(gf))
    plt.title('Plot of $P(t)$ with initial pop {}'.format(init_pop))
    plt.xlabel('Time in years')
    plt.ylabel('Population in thousands')
    plt.legend()
