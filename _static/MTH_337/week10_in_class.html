In [1]:
def the_straightforward_way(arr,k):
    m,n = np.shape(arr)
    for i in range(m):
        for j in range(n):
            if arr[i,j] == k:
                arr[i,j] = 1
            else:
                arr[i,j] = 0
    return arr

def the_boolean_way(arr,k):
    arr = arr == k
    arr = np.array(arr,dtype = int)
    return arr
