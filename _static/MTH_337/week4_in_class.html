import random
def fun1(n):
    heads = 0
    for i in range(n):
        if random.randint(1,2) == 1:
            heads += 1
    return heads

def basic_population(T,initial_pop,growth_factor):
    years = [year for year in range(T+1)]
    population = [initial_pop]
    for i in range(T):
        population.append(population[i]*growth_factor)
    #plt.plot(years,population)

    return population
